#include <Arduino.h>

// Pin tanımlamaları
const int potPin = 34; // Potansiyometrenin bağlı olduğu pin (örneğin, GPIO 34)
const int ledPin = 2;  // LED'in bağlı olduğu pin (örneğin, GPIO 2)

void setup() {
  pinMode(ledPin, OUTPUT); // LED pinini çıkış olarak ayarla
  Serial.begin(9600);      // Seri bağlantıyı başlat
}

void loop() {
  int sensorValue = analogRead(potPin); // Potansiyometre değerini oku
  int brightness = map(sensorValue, 0, 4095, 0, 255); // Potansiyometre değerini 0-4095 aralığından 0-255 aralığına dönüştür
  
  analogWrite(ledPin, brightness); // LED'in parlaklığını ayarla
  
  Serial.print("Potansiyometre Değeri: ");
  Serial.print(sensorValue);
  Serial.print(" | Parlaklık: ");
  Serial.println(brightness);

  delay(100); // Kısa bir bekleme ekleyin (değeri düşürerek LED'in daha yumuşak bir şekilde yanmasını sağlayabilirsiniz)
}
