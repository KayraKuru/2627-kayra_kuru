#include <Arduino.h>

// Pin tanımlamaları
const int OLED_RESET = -1; // Eğer reset pini kullanılmıyorsa -1 olarak ayarlanır
const int OLED_SDA = 21;   // OLED ekranın SDA pinini ESP32'nin SDA pinine bağlayın (örneğin, GPIO 21)
const int OLED_SCL = 22;   // OLED ekranın SCL pinini ESP32'nin SCL pinine bağlayın (örneğin, GPIO 22)

#include <Wire.h>   // I2C kütüphanesi ekleme
#include <Adafruit_GFX.h>  // Grafik kütüphanesi ekleme
#include <Adafruit_SSD1306.h> // SSD1306 OLED ekran kütüphanesi ekleme

#define SCREEN_WIDTH 128  // OLED ekran genişliği
#define SCREEN_HEIGHT 64  // OLED ekran yüksekliği

// OLED ekran nesnesini oluşturma
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

void setup() {
  // Seri bağlantıyı başlatma
  Serial.begin(9600);

  // OLED ekranı başlatma
  if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) { 
    Serial.println(F("SSD1306 ekran bulunamadı!"));
    while (true);
  }

  // Ekranı temizleme
  display.clearDisplay();
}

void loop() {
  // Ekranı temizleme
  display.clearDisplay();

  // Mesajı yazdırma
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(0,0);
  display.println("Merhaba, ESP32!");
  
  // Ekranı güncelleme
  display.display();
  
  delay(1000); // 1 saniye bekleme
